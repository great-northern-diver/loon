184  proc recursive_divide {num_set} {
185  set new_num_set [list]
186  set out [list]
187 
188  foreach nums $num_set {
189  set n [llength $nums]
190  if {$n eq 1} {
191  lappend out $nums
192  } else {
193  if {[expr {$n%2}] eq 1} {
194  set i [expr {($n+1)/2 - 1}]
195  lappend out [lindex $nums $i]
196  lappend new_num_set [lrange $nums 0 [expr {$i-1}]]
197  lappend new_num_set [lrange $nums [expr {$i+1}] end]
198  } else {
199  lappend out [lindex $nums 0]
200  lappend new_num_set [lrange $nums 1 end]
201  }
202  }
203  }

205  #puts "new_num_set=$new_num_set"
206  if {[llength $new_num_set] eq 0} {
207  return $out
208  } else {
209  return [concat $out [recursive_divide $new_num_set]]

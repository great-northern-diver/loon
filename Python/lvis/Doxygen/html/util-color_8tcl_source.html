17  # rgb2dec --
18  #
19  # Turns #rgb into 3 elem list of decimal vals.
20  #
21  # Arguments:
22  # c The #rgb hex of the color to translate
23  # Results:
24  # Returns a #RRGGBB or #RRRRGGGGBBBB color
25  #
26  proc rgb2dec c {
27  set c [string tolower $c]
28  if {[regexp {^#([0-9a-f])([0-9a-f])([0-9a-f])$} $c x r g b]} {
29  # double'ing the value make #9fc == #99ffcc
30  scan "$r$r $g$g $b$b" "%x %x %x" r g b
31  } else {
32  if {![regexp {^#([0-9a-f]+)$} $c junk hex] || \
33  [set len [string length $hex]]>12 || $len%3 != 0} {
34  return -code error "bad color value \"$c\""
35  }
36  set len [expr {$len/3}]
37  scan $hex "%${len}x%${len}x%${len}x" r g b
38  }
39  return [list $r $g $b]
40  }

42  # dec2rgb --
43  #
44  # Takes a color name or dec triplet and returns a #RRGGBB color.
45  # If any of the incoming values are greater than 255,
46  # then 16 bit value are assumed, and #RRRRGGGGBBBB is
47  # returned, unless $clip is set.
48  #
49  # Arguments:
50  # r red dec value, or list of {r g b} dec value or color name
51  # g green dec value, or the clip value, if $r is a list
52  # b blue dec value
53  # clip Whether to force clipping to 2 char hex
54  # Results:
55  # Returns a #RRGGBB or #RRRRGGGGBBBB color
56  #
57  proc dec2rgb {r {g 0} {b UNSET} {clip 0}} {
58  if {![string compare $b "UNSET"]} {
59  set clip $g
60  if {[regexp {^-?(0-9)+$} $r]} {
61  foreach {r g b} $r {break}
62  } else {
63  foreach {r g b} [winfo rgb . $r] {break}
64  }
65  }
66  set max 255
67  set len 2
68  if {($r > 255) || ($g > 255) || ($b > 255)} {
69  if {$clip} {
70  set r [expr {$r>>8}]; set g [expr {$g>>8}]; set b [expr {$b>>8}]
71  } else {
72  set max 65535
73  set len 4
74  }
75  }
76  return [format "#%.${len}X%.${len}X%.${len}X" \
77  [expr {($r>$max)?$max:(($r<0)?0:$r)}] \
78  [expr {($g>$max)?$max:(($g<0)?0:$g)}] \
79  [expr {($b>$max)?$max:(($b<0)?0:$b)}]]
80  }

82  # shade --
83  #
84  # Returns a shade between two colors
85  #
86  # Arguments:
87  # orig start #rgb color
88  # dest #rgb color to shade towards
89  # frac fraction (0.0-1.0) to move $orig towards $dest
90  # Results:
91  # Returns a shade between two colors based on the
92  #
93  proc shade {orig dest frac} {
94  if {$frac >= 1.0} { return $dest } elseif {$frac <= 0.0} { return $orig }
95  foreach {origR origG origB} [rgb2dec $orig] \
96  {destR destG destB} [rgb2dec $dest] {
97  set shade [format "\#%02x%02x%02x" \
98  [expr {int($origR+double($destR-$origR)*$frac)}] \
99  [expr {int($origG+double($destG-$origG)*$frac)}] \
100  [expr {int($origB+double($destB-$origB)*$frac)}]]
101  return $shade
102  }
103  }

105  # complement --
106  #
107  # Returns a complementary color
108  # Does some magic to avoid bad complements of grays
109  #
110  # Arguments:
111  # orig start #rgb color
112  # Results:
113  # Returns a complement of a color
114  #
115  proc complement {orig {grays 1}} {
116  foreach {r g b} [rgb2dec $orig] {break}
117  set R [expr {(~$r)%256}]
118  set G [expr {(~$g)%256}]
119  set B [expr {(~$b)%256}]
120  if {$grays && abs($R-$r) < 32 && abs($G-$g) < 32 && abs($B-$b) < 32} {
121  set R [expr {($r+128)%256}]
122  set G [expr {($g+128)%256}]
123  set B [expr {($b+128)%256}]
124  }
125  return [format "\#%02x%02x%02x" $R $G $B]
126  }

128  # hsv2rgb --
129  #
130  # Convert hsv to rgb
131  #
132  # Arguments:
133  # h hue
134  # s saturation
135  # v value
136  # Results:
137  # Returns an rgb triple from hsv
138  #
139  proc hsv2rgb {h s v} {
140  if {$s <= 0.0} {
141  # achromatic
142  set v [expr {int($v)}]
143  return "$v $v $v"
144  } else {
145  set v [expr {double($v)}]
146  if {$h >= 1.0} { set h 0.0 }
147  set h [expr {6.0 * $h}]
148  set f [expr {double($h) - int($h)}]
149  set p [expr {int(256 * $v * (1.0 - $s))}]
150  set q [expr {int(256 * $v * (1.0 - ($s * $f)))}]
151  set t [expr {int(256 * $v * (1.0 - ($s * (1.0 - $f))))}]
152  set v [expr {int(256 * $v)}]
153  switch [expr {int($h)}] {
154  0 { return "$v $t $p" }
155  1 { return "$q $v $p" }
156  2 { return "$p $v $t" }
157  3 { return "$p $q $v" }
158  4 { return "$t $p $v" }
159  5 { return "$v $p $q" }
160  }
161  }
162  }

164  proc hls2rgb {h l s} {
165  # Posted by frederic.bonnet@ciril.fr
166  # h, l and s are floats between 0.0 and 1.0, ditto for r, g and b
167  # h = 0 => red
168  # h = 1/3 => green
169  # h = 2/3 => blue
170 
171  set h6 [expr {($h-floor($h))*6}]
172  set r [expr { $h6 <= 3 ? 2-$h6
173  : $h6-4}]
174  set g [expr { $h6 <= 2 ? $h6
175  : $h6 <= 5 ? 4-$h6
176  : $h6-6}]
177  set b [expr { $h6 <= 1 ? -$h6
178  : $h6 <= 4 ? $h6-2
179  : 6-$h6}]
180  set r [expr {$r < 0.0 ? 0.0 : $r > 1.0 ? 1.0 : double($r)}]
181  set g [expr {$g < 0.0 ? 0.0 : $g > 1.0 ? 1.0 : double($g)}]
182  set b [expr {$b < 0.0 ? 0.0 : $b > 1.0 ? 1.0 : double($b)}]
183 
184  set r [expr {(($r-1)*$s+1)*$l}]
185  set g [expr {(($g-1)*$s+1)*$l}]
186  set b [expr {(($b-1)*$s+1)*$l}]
187  return [list $r $g $b]
188  }

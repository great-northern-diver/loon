335 ## origin can be any bin start
336 ## used algorithm from
337 ## http://stackoverflow.com/questions/2387916/looking-for-a-histogram-binning-algorithm-for-decimal-data
338 ## as a starting point
339 proc ::loon::binningalg {xvals binwidth origin active\
340  selected color {binningtype frequency} {binalgRun ""}} {
341 
342  if {$binwidth <= 0} {
343  error "bindiwth must be a positive number."
344  }
345 
346 
347  ## bins data structure
348  ##
349  ## binid: n dimensional list with binid for each point
350  ##
351  ## ...,-1,0,1,... (binids):
352  ## count
353  ## all
354  ## selected
355  ## <color>
356  ## points
357  ## all
358  ## selected
359  ## <color>
360  ## x0
361  ## x1
362 
363  set bins [dict create bin [dict create] binid {}]
364 
365  if {[llength $binalgRun] > 0} {
366  dict set bins binalgRun $binalgRun
367  }
368 
369  set i 0
370  foreach x $xvals a $active col $color sel $selected {
371  if {$a} {
372  set binid [expr {int(floor(($x-$origin)/$binwidth))}]
373  if {![dict exists $bins bin $binid]} {
374  dict set bins bin $binid\
375  [dict create count [dict create all 0 selected 0]\
376  points [dict create all {} selected {}]]
377  }
378 
379  dict with bins bin $binid {dict incr count all}
380  dict with bins bin $binid {dict lappend points all $i}
381 
382  if {$sel} {
383  dict with bins bin $binid {dict incr count selected}
384  dict with bins bin $binid {dict lappend points selected $i}
385  } else {
386  dict with bins bin $binid {dict incr count $col}
387  dict with bins bin $binid {dict lappend points $col $i}
388  }
389 
390  } else {
391  set binid NA
392  }
393 
394  incr i
395  dict lappend bins binid $binid
396  }
397 
398  ## now get x0 and x1 for each bin
399  ## need to get both as otherwise i might get pixel errors
400  dict with bins {dict for {id value} $bin {
401  dict set bin $id x0 [expr {$origin + $id * $binwidth}]
402  dict set bin $id x1 [expr {$origin + ($id + 1)* $binwidth}]
403  }}
404 
405 
406  switch -- $binningtype {
407  density {
408  if {[::loon::listfns::any $active]} {
409 
410  set n 0
411  foreach a $active {
412  if {$a} {incr n}
413  }
414 
415  set area [expr {$binwidth * $n}]
416 
417  ## divide by area
418  dict with bins {
419  dict for {id value} $bin {
420  dict with value {
421  dict for {type num} $count {
422  dict set bin $id count $type\
423  [expr {double($num)/$area}]
424  }
425  }
426  }
427  }
428  }
429  }
430  }
431  return $bins
432 }

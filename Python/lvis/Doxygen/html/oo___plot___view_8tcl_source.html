151  method plotUpdate {Events} {
152 
153  ## A couple of events in are not of interest
154  ## for the decendents of Plot_View
155  if {$Events eq "selected" || $Events eq "color" ||\
156  $Events eq "size" || $Events eq "active"} {
157  return
158  }
159 
160  ## Copy events in a dict
161  set events [dict create]
162  foreach e $Events {
163  dict set events $e 1
164  }
165 
166  if {[dict exists $events destroy]} {
167  ## Model got destroyed
168  my ModelDestroyed
169  return
170  }
171 
172  ## For speed if only zoom and pan
173  set nevents [llength $Events]
174  set numHits 0
175 
176  ## keep numHits to improve on performance
177 
178  set oldPanX [$map getPanX]
179  if {[dict exists $events "panX"]} {
180  incr numHits
181  set newPanX [set $panX_var]
182  $map setPanX $newPanX
183  } else {
184  set newPanX $oldPanX
185  }
186 
187  set oldPanY [$map getPanY]
188  if {[dict exists $events "panY"]} {
189  incr numHits
190  set newPanY [set $panY_var]
191  $map setPanY $newPanY
192  } else {
193  set newPanY $oldPanY
194  }
195 
196  if {$numHits >= $nevents} {
197  my updatePan [expr {$oldPanX - $newPanX}]\
198  [expr {$oldPanY - $newPanY}]
199  return
200  }
201 
202 
203  set oldZoomX [$map getZoomX]
204  if {[dict exists $events "zoomX"]} {
205  incr numHits
206  set newZoomX [set $zoomX_var]
207  $map setZoomX $newZoomX
208  }
209 
210  set oldZoomY [$map getZoomY]
211  if {[dict exists $events "zoomY"]} {
212  incr numHits
213  set newZoomY [set $zoomY_var]
214  $map setZoomY $newZoomY
215  }
216 
217  if {$numHits > 0} {
218  my updateZoomPan $oldPanX $oldPanY $oldZoomX $oldZoomY
219  }

269  method aspect {args} {
270 
271  set rX [set $deltaX_var]
272  set rY [set $deltaY_var]

275 
276  if {[set $swap_var]} {

279  } else {

281  set dy [$map getPlotHeight]
282  }
283 
284  switch -- [llength $args] {
285  0 {
286  set alpha [expr {double($dy)*$zY/$rY*$rX/($dx*$zX)}]
287  return $alpha
288  }
289  1 {
290  ## set aspect ratio
291  ## change zoomY
292  set alpha $args
293  if {![string is double $alpha] || $alpha <= 0 } {
294  error "alpha is \"$alpha\" but needs to be a strictly positive number"
295  }
296 
297  set zoomY [expr {double($alpha)*$rY/$dy*($dx*$zX)/$rX}]
298

300  uplevel #0 [list $plotModel configure -zoomY $zoomY]
301  }
302 
303  }

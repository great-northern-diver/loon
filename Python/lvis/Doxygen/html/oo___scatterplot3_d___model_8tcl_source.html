182  method project {xs ys zs rotationCenter} {
183  my variable rotate3DX rotate3DY
184 
185  set cosAX [expr {cos($rotate3DX)}]
186  set cosAY [expr {cos($rotate3DY)}]
187  set sinAX [expr {sin($rotate3DX)}]
188  set sinAY [expr {sin($rotate3DY)}]
189 
190  # Rotation Matrices
191  set Rx [list [list 1.0 0.0 0.0] [list 0.0 $cosAX [expr {-1 * $sinAX}]] [list 0.0 $sinAX $cosAX]]
192  set Ry [list [list $cosAY 0.0 $sinAY] [list 0.0 1.0 0.0] [list [expr {-1 * $sinAY}] 0.0 $cosAY]]
193 
194  set R [::loon::listfns::matmul $Rx $Ry]
195 
196  set xProjected {}
197  set yProjected {}
198  set zProjected {}
199  foreach xe $xs ye $ys ze $zs {
200  # Transform to coordinates relative to rotation origin
201  set xShifted [expr {$xe - [lindex $rotationCenter 0]}]
202  set yShifted [expr {$ye - [lindex $rotationCenter 1]}]
203  set zShifted [expr {$ze - [lindex $rotationCenter 2]}]
204 
205  # Rotate
206  set xTemp [::loon::listfns::dot [lindex $R 0] [list $xShifted $yShifted $zShifted]]
207  set yTemp [::loon::listfns::dot [lindex $R 1] [list $xShifted $yShifted $zShifted]]
208  set zTemp [::loon::listfns::dot [lindex $R 2] [list $xShifted $yShifted $zShifted]]
209 
210  # Transform back to original coordinates
211  lappend xProjected [expr {$xTemp + [lindex $rotationCenter 0]}]
212  lappend yProjected [expr {$yTemp + [lindex $rotationCenter 1]}]
213  lappend zProjected [expr {$zTemp + [lindex $rotationCenter 2]}]
214  }
215  return [dict create x $xProjected y $yProjected z $zProjected]
216  }

61  lappend thisrow $pr $pg $pb
62  incr xtot $mx
63  incr nx
64  }
65

68 
69  # Calculate ratios to use
70

72  set rn $xtot
73  set rp [expr {$mx - $xtot}]
74 
75  # This section covers shrinking an image where
76  # more than 1 source pixel may be required to
77  # define the destination pixel
78 
79  set xr 0
80  set xg 0
81  set xb 0
82

84  incr xr $r
85  incr xg $g
86  incr xb $b
87 
88  set xtot [expr {$xtot - $newx}]
89  incr x
90  foreach {r g b} [$src get $x $y] {break}
91

95

97  set tg [expr {int( ($rn*$g + $xg + $rp*$pg) / $mx)}]
98  set tb [expr {int( ($rn*$b + $xb + $rp*$pb) / $mx)}]
99

101  if {$tg > 255} {set tg 255}
102  if {$tb > 255} {set tb 255}

112  set pg $g
113  set pb $b
114  }

119  lappend thisrow $r $g $b

151 
152  while { $ytot > $newy } {
153

155  incr y

157  }

162 
163  set tr [expr {int( ($rn*$r + $rp*$pr) / $my)}]
164  set tg [expr {int( ($rn*$g + $rp*$pg) / $my)}]
165  set tb [expr {int( ($rn*$b + $rp*$pb) / $my)}]
